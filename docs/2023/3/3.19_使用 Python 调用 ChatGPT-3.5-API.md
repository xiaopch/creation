## 使用 Python 调用 ChatGPT-3.5-API

```
import openai
##pip install -U openai -i pypi.douban.com/simple
import os
##from zhconv import convert
from datetime import datetime


# 目前需要设置代理才可以访问 api
os.environ["http_proxy"] = "http://127.0.0.1:7890"
os.environ["https_proxy"] = "http://127.0.0.1:7890"
# 填写注册OpenAI接口账号时获取的 OpenAI API Key
openai.api_key = '########'

# 提问
prompt = '请问怎么设计chagpt的prompt'

# 访问OpenAI接口
response = openai.ChatCompletion.create(
    model='gpt-3.5-turbo',
    messages=[
        {'role': 'system', 'content': '你是小朋的私人智能助手。'},
        # {'role': 'user', 'content': '谁赢得了2020年的世界职业棒球大赛?'},
        # {'role': 'assistant', 'content': '洛杉矶道奇队在2020年赢得了世界职业棒球大赛冠军。'},
        {'role': 'user', 'content': prompt}
    ]
)

# 返回信息

resText = response.choices[0].message.content

print(resText)

##将返回的内容保存到markdown文件

time_str = datetime.now().strftime('%Y_%m_%d_%H%M%S.')

filename = time_str+prompt+'.md'
print(filename)
with open(filename,'w') as fh:
    fh.write('## {}\n'.format(prompt))
    fh.write('```\n')
    fh.write(resText+'\n')
    fh.write('```')

```
