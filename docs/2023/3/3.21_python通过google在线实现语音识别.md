## python通过google在线实现语音识别

```
import speech_recognition as sr

##从终端安装 SpeechRecognition
##使用命令：pip3 install SpeechRecognition
##pip install pyaudio

print(sr.__version__)


import os
os.environ["HTTP_PROXY"] = "http://127.0.0.1:7890"
os.environ["HTTPS_PROXY"] = "http://127.0.0.1:7890"


r = sr.Recognizer()
 
test = sr.AudioFile('a.wav')
 
with test as source:
##    audio = r.record(source)
    audio = r.record(source, offset=1, duration=3)
 
result = r.recognize_google(audio, language='zh-CN', show_all= True)
print(result)

```
