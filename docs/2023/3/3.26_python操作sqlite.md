## python操作sqlite

```
#!/usr/bin/env python
#-*-coding:utf-8 -*-

import sqlite3
import os, sys

db_name = "./sqlite.db";

# 用:memory:创建的每个数据库都是不同的。
# 创建内存数据库
# db_name = ":memory:";

table_name = 'chatgpt'

# 创建表格
def initialize(conn):
    global table_name
    c = conn.cursor()
    sql = "CREATE TABLE IF NOT EXISTS {} (_id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,question CHAR(100) NOT NULL, answer TEXT,category  INT DEFAULT 0,note  CHAR(100) DEFAULT '')".format(table_name)
    # print(sql)
    c.execute(sql);
    conn.commit();

# print("显示所有记录")
def display(conn):
    c = conn.cursor();
    c.execute("SELECT * FROM {}".format(table_name));
    print(c.fetchall());

conn = sqlite3.connect(db_name);
initialize(conn);
print("插入记录")
c = conn.cursor();
c.execute("INSERT INTO {} (question,answer,category,note) VALUES (?,?,?,?)".format(table_name), ("name","Alice",1,''));
conn.commit();
# print("删除记录")

# display(conn);

print("更新记录")
c = conn.cursor();
c.execute("UPDATE chatgpt SET answer = ? WHERE _id = ?", ('answer',2 ));
conn.commit();
display(conn);

conn.close();
```
