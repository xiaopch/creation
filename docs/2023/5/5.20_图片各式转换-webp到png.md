## 图片各式转换-webp到png

```
import os
import shutil

import hashlib
from PIL import Image, ImageOps

def getmd5(filename):
    # 接收文件路径，返回文件md5值
    with open(filename, 'rb') as f:
        data = f.read()
    file_md5 = hashlib.new("md5", data).hexdigest()
    return file_md5

imagePath_src = './Midjourney Showcase_files/' # input('请输入图片所在的文件夹路径：')
imagePath_dest = './Midjourney Showcase/' # input('请输入图片所在的文件夹路径：')
if not os.path.exists(imagePath_dest):
    os.mkdir(imagePath_dest)
# 遍历文件，将所有文件的路径放入一个列表里
def file_walk(root_path,ext='.webp'):
    file_list = []
    for dirPath, dirNames, fileNames in os.walk(root_path):
        for fileName in fileNames:
            if fileName.lower().endswith(ext):
                filePath = os.path.join(dirPath, fileName)
                file_list.append(filePath)
    return file_list

def webp2png(file_list):
    for srcImagePath in file_list:
        image = Image.open(srcImagePath)
        image = ImageOps.exif_transpose(image)
        # dstImagePath = os.path.splitext(srcImagePath)[0] + '.png'
        dstImagePath =imagePath_dest + getmd5(srcImagePath) + '.png'

        image.save(dstImagePath)
        print('%s ---> %s' % (srcImagePath, dstImagePath))
        os.remove(srcImagePath)
        # if os.path.exists(dstImagePath):
        #     os.remove(srcImagePath)
        #     continue



if __name__ == '__main__':

    image_list = file_walk(imagePath_src,'.webp')
    # print(image_list)
    webp2png(image_list)
```
