## python实现批量文件重命名

```
#!/usr/bin/env python
#coding=utf-8
import os,os.path
import shutil,string
import time
def renameTool(dirname,str_origin,str_dest):
    iNum=0
    if os.path.exists(dirname):
        for root, dirs, files in os.walk(dirname,True):
            for oldname in files:
                oldname=os.path.join(root,oldname)

                newname = oldname.replace(str_origin,str_dest)
                os.rename(oldname,newname)   #用os模块中的rename方法对文件改名
                print(oldname,'======>',newname)
    #             iNum=iNum+1

if __name__ == '__main__':
    dirname = './'
    str_origin = '''（）'''
    str_dest = ''''''
    renameTool(dirname,str_origin,str_dest)
```
