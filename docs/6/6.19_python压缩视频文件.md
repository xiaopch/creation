#!/usr/bin/python
# -*- coding: UTF-8 -*-
import os

for root, dirs, files in os.walk(".", topdown=False):
    for name in files:
        try:
            file_path = os.path.join(root, name)
            if file_path.endswith('.mp4'):

                input_file = file_path
                output_file = file_path.replace('.mp4','.small.mp4')
                print(output_file)
                cmd = f"ffmpeg -threads 16 -i {input_file} -c:v libx265 -crf 25 -vf scale=800:-1 -r 10  -b:a 128k -b:v 128k  -y {output_file}"
                os.system(cmd)
        except:
            print('error!')
