## gradio实现chatgpt对话

```
import gradio as gr
import random
import time
import openai
from datetime import datetime

import os
##设置代理
os.environ["HTTP_PROXY"] = "http://127.0.0.1:7890"
os.environ["HTTPS_PROXY"] = "http://127.0.0.1:7890"
openai.api_key = "sk-***"  # 更换为自己的密钥

msg_list = []

prompt_init = """请输入您的问题."""
bot_message =''

# 获取chatgpt的回答
def openai_create(prompt):
    completion = openai.ChatCompletion.create(
    model="gpt-3.5-turbo",
    messages=[
    {"role": "system", "content": "你是一名个人智能助理。"},
    {"role": "user", "content": f"{prompt}"}
      ]
    )
    # print(completion)
    return completion.choices[0].message.content

#保存对话历史
def save_result(msg):
    global msg_list
    filename = datetime.now().strftime('History_%Y_%m_%d_%H%M%S.md')#%Y_%m_%d_%H%M%S
    with open(filename,'w',encoding='utf-8') as fh:
        # fh.write('## {}\n'.format(md_file_name))
        # print(chat.messages)
        # line = '```\n{}```\n'.format(i['role']+'\n')
        for i in msg_list:
            # print(i['role']) # 结果 pengjunlee
            if i['role'] == 'user':
                fh.write('> [Prompt]{}'.format(i['content']+'\n'))
            else:
                fh.write('\n{}\n'.format(i['content']+'\n'))
    return 'File {} Saved!'.format(filename)

# gradio block
with gr.Blocks() as demo:
    gr.Markdown("""<h1><center>ChatGPT对话系统</center></h1>""")

    chatbot = gr.Chatbot(label="History:")
    msg = gr.Textbox(placeholder=prompt_init,label="Prompt:")

    with gr.Row():
        send_btn = gr.Button("SEND")
        Save_Btn = gr.Button("Save")
        Reset_Btn = gr.Button("Reset")
        Format_Btn = gr.Button("Format")

    gr.Markdown("""Message:""")
    # output = gr.Markdown()
    output = gr.Code()
    # chatbot 响应
    def respond(message, chat_history):
        global msg_list
        global bot_message
        # msg.update=message
        msg_list.append({"role": "user", "content": message})
        bot_message = openai_create(msg_list)
        msg_list.append({"role": "“assistant”", "content": bot_message})
        chat_history.append((message, bot_message))
        time.sleep(1)
        return "", chat_history
    #格式化显示当前answer
    def format_answer():
        global bot_message
        return bot_message

    def reset_dialog():
        msg_list = []
        return 'dialog reset'

    Reset_Btn.click(lambda: None, None, chatbot, queue=False)
    Reset_Btn.click(reset_dialog, None, output)
    Save_Btn.click(fn=save_result, inputs=msg, outputs=output)
    send_btn.click(respond, inputs=[msg, chatbot], outputs=[msg, chatbot])
    msg.submit(respond, [msg, chatbot], [msg, chatbot])
    Format_Btn.click(format_answer, None, output)

demo.launch(inbrowser=False, inline=False, share=True)

import webbrowser
webbrowser.open('http://127.0.0.1:7860')
```
