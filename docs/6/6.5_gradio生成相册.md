## gradio生成相册

```
import gradio as gr
import glob
import os

image_dir = '428'
image_format = 'png'

#生成image数据响应
def get_image():
    images = [(i,os.path.splitext(i)[0].split('\\')[-1]) for i in glob.glob(r'.\{}\*.{}'.format(image_dir,image_format))]
    return images

#接口创建函数
with gr.Blocks() as demo:
    btn = gr.Button("Show image").style(full_width=False)
    gallery = gr.Gallery(label="show images", show_label=False, elem_id="gallery").style(height="auto",preview=True)
    btn.click(fn=get_image,outputs=gallery)
    demo.title = "Gallery"

demo.launch()

import webbrowser
webbrowser.open('http://127.0.0.1:7860')
```
