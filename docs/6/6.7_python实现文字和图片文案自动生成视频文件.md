## python实现文字和图片文案自动生成视频文件

#### python代码如下:

```
# from moviepy.editor import *
import os
from glob import glob
from PIL import Image, ImageOps
import subprocess
from xml.etree.ElementTree import parse

tts_executor = None

#图片格式转化
def webp2png(file_list):
    for srcImagePath in file_list:
        image = Image.open(srcImagePath)
        image = ImageOps.exif_transpose(image)
        dstImagePath = os.path.splitext(srcImagePath)[0] + '.png'
        # dstImagePath =imagePath_dest + getmd5(srcImagePath) + '.png'

        image.save(dstImagePath)
        print('%s ---> %s' % (srcImagePath, dstImagePath))
        # os.remove(srcImagePath)

#将图像和文字组合成mp4视频
def image_text_to_mp4(img_file,content,video_file,spk_id=171):
    global tts_executor
    audio_file_output = 'audio.wav'

    output_file = video_file
    wav_file = tts_executor(
        text=content,
        output=audio_file_output,
        am='fastspeech2_aishell3', #tacotron2_csmsc
        am_config=None,
        am_ckpt=None,
        am_stat=None,
        spk_id=spk_id,  #171 48 10 58 60
        phones_dict=None,
        tones_dict=None,
        speaker_dict=None,
        voc='hifigan_csmsc', # mb_melgan_csmsc wavernn_csmsc hifigan_csmsc
        voc_config=None,
        voc_ckpt=None,
        voc_stat=None,
        lang='zh',
        cpu_threads=2)

    # 调用ffmpeg命令进行视频合并
    cmd = f'ffmpeg -framerate 1 -f image2 -loop 1 -y  -i {img_file} -i {audio_file_output} -movflags +faststart -shortest -fflags +shortest -r 25  {output_file}'
    print(cmd)
    # 执行命令
    subprocess.call(cmd, shell=True)

if __name__ == '__main__':

    # 将图片格式转换成png
    image_list = glob('./*.jpg')+glob('./*.webp')
    print(image_list)
    webp2png(image_list)
    os.system('del *.webp')
    os.system('del *.jpg')

    # 引入文字转语音库
    from paddlespeech.cli.tts import TTSExecutor
    tts_executor = TTSExecutor()

    # 开始读取xml文件中的配置信息
    root = parse('data.xml').getroot()
    # print(root.tag)
    # 文章标题和生成TTS的声音
    title= root.findall('title')[0].text
    speak_id= int(root.findall('speak_id')[0].text)
    print(f'speak_id:{speak_id}')
    # 删除过去生成的视频文件
    os.system('del *.mp4')

    video_file_list = []
    # 开始生成mp4视频
    # 查找XML文档中的元素
    index = 1
    for child in root.findall('content'):
        # 获取元素的属性值或文本内容
        text = child.text
        # print(f'text:{text}')
        img_file=child.attrib['image']
        # print(img_file)
        # print(f'attrib:{child.attrib}')
        video_file = '{}.mp4'.format(index)
        video_file_list.append(video_file)
        print(img_file,text,video_file)
        image_text_to_mp4(img_file,text,video_file,speak_id)
        index = index+1

    #生成合并的视频的文件
    video_file_list_for_ffmpeg = 'video_file_list_for_ffmpeg.txt'
    with open(video_file_list_for_ffmpeg,'w') as f:
        for file in video_file_list:
            f.write('file ' + '\''+file+'\'\n')
    # 利用标题生成视频文件名
    out_file_name = '{}.mp4'.format(title)
    print('video file:{}'.format(out_file_name))


    # 调用系统命令行合并mp4文件
    os.system('ffmpeg.exe -f concat -safe 0 -i {} -c copy -y {}'.format(video_file_list_for_ffmpeg,out_file_name))

    #删除临时文件
    os.system('del audio.wav')
    os.system(f'del {video_file_list_for_ffmpeg}')

    for file in video_file_list:
        os.system('del {}'.format(file))
```
